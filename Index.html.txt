        function isValidCoordinatePair(coords) {
            return Array.isArray(coords)
                && coords.length === 2
                && Number.isFinite(coords[0])
                && Number.isFinite(coords[1])
                && coords[0] >= -180
                && coords[0] <= 180
                && coords[1] >= -90
                && coords[1] <= 90;
        }

        function calculateDistanceMeters(p1, p2) {
            const R = 6371e3;
            const φ1 = p1[1] * Math.PI/180;
            const φ2 = p2[1] * Math.PI/180;
            const Δφ = (p2[1]-p1[1]) * Math.PI/180;
            const Δλ = (p2[0]-p1[0]) * Math.PI/180;

            const a = Math.sin(Δφ/2) * Math.sin(Δφ/2) +
                      Math.cos(φ1) * Math.cos(φ2) *
                      Math.sin(Δλ/2) * Math.sin(Δλ/2);
            const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));

            return R * c;
        }

        function smoothCoordinatePair(oldCoords, newCoords) {
            if (!oldCoords || !newCoords) return newCoords;
            return [
                oldCoords[0] + (newCoords[0] - oldCoords[0]) * GPS_SMOOTHING_FACTOR,
                oldCoords[1] + (newCoords[1] - oldCoords[1]) * GPS_SMOOTHING_FACTOR
            ];
        }

        function calculateBearingAngle(p1, p2, fallbackBearing = currentVehicleHeading) {
            if (!isValidCoordinatePair(p1) || !isValidCoordinatePair(p2)) return fallbackBearing;
            const samePoint = Math.abs(p1[0] - p2[0]) < 0.000001 && Math.abs(p1[1] - p2[1]) < 0.000001;
            if (samePoint) return fallbackBearing;

            const lng1 = p1[0] * Math.PI / 180;
            const lat1 = p1[1] * Math.PI / 180;
            const lng2 = p2[0] * Math.PI / 180;
            const lat2 = p2[1] * Math.PI / 180;

            const dLng = lng2 - lng1;
            const y = Math.sin(dLng) * Math.cos(lat2);
            const x = Math.cos(lat1) * Math.sin(lat2) - Math.sin(lat1) * Math.cos(lat2) * Math.cos(dLng);

            const bearing = Math.atan2(y, x) * 180 / Math.PI;
            return Number.isFinite(bearing) ? (bearing + 360) % 360 : fallbackBearing;
        }

            // ── Interaction ──────────────────────────────────────────────

            _laneAtEvent(e) {
                const rect = this.canvas.getBoundingClientRect();
                const x = (e.clientX - rect.left) * (this.W / rect.width);
                const y = (e.clientY - rect.top)  * (this.H / rect.height);
                // Cast the point to road coordinates at its depth
                const t = Math.max(0, Math.min(0.9, (this.H - y) / (this.H - 30)));
                // Approximate inverse of _project on x
                const baseHalf = this.W * 0.62;
                const topHalf  = this.W * 0.13;
                const halfW = baseHalf + (topHalf - baseHalf) * t;
                const bend  = this._effectiveCurve() * t * t * (this.W * 0.55);
                const cx = this.W / 2 + bend + this.mapBearingOffset * 40;
                const u = ((x - cx) / (2 * halfW)) + 0.5;
                if (u < 0 || u > 1) return null;
                return Math.floor(u * this.numLanes);
            }

            _onCanvasClick(e) {
                const i = this._laneAtEvent(e);
                if (i == null || i < 0 || i >= this.numLanes) return;
                this.userPickedLane = (this.userPickedLane === i) ? null : i;
            }

            _onCanvasHover(e) {
                this.hoverLane = this._laneAtEvent(e);
            }

            // ── Maneuver / lane computation ─────────────────────────────

            _dirFor(type, mod) {
                if (mod === 'uturn' || mod === 'u-turn') return 'uturn';
                if (mod === 'sharp left')                return 'sharp-left';
                if (mod === 'slight left')               return 'slight-left';
                if (mod === 'left')                      return 'left';
                if (mod === 'sharp right')               return 'sharp-right';
                if (mod === 'slight right')              return 'slight-right';
                if (mod === 'right')                     return 'right';
                if (type === 'arrive')                   return 'arrive';
                return 'straight';
            }

            _computeManeuver(type, mod) {
                this.maneuverDir = this._dirFor(type, mod);

            _computeLanes(step, type, mod) {
                // 1) Prefer real router lane data when present
                if (Array.isArray(step.lanes) && step.lanes.length) {
                    this.numLanes = step.lanes.length;
                    this.lanes = step.lanes.map(l => ({
                        dirs:   Array.isArray(l.indications) ? l.indications : [l.direction || 'straight'],
                        active: !!l.valid
                    }));
                    return;
                }

                // 2) Geometry-driven heuristic fallback.
                //    Combines: turn angle, step length, maneuver type,
                //    road class inference, and whether we're on a ramp/fork.

                // ── Turn angle from actual coordinates (fallback: modifier)
                const turnDeg = this._estimateTurnDeg(step);

                // ── Guess how many lanes this road plausibly has.
                // Motorways / long straight steps → 3-4 lanes.
                // Trunk / arterial / medium steps → 3 lanes.
                // Short urban steps → 2 lanes.
                const km = (step.distance || 0) / 1000;
                let n;
                if (this.isRamp)                          n = 2;
                else if (step.roadClass === 'motorway')   n = 4;
                else if (step.roadClass === 'trunk')      n = 3;
                else if (step.roadClass === 'primary')    n = 3;
                else if (km > 1.5)                        n = 3;
                else if (km > 0.4)                        n = 2 + (turnDeg < 25 ? 1 : 0);
                else                                      n = 2;
                n = Math.max(2, Math.min(5, n));
                this.numLanes = n;

                // ── Multi-exit ramp: dedicated exit lanes on the side.
                if (this.isRamp) {
                    const rightSide = !/left/.test(this.maneuverDir);
                    this.rampLanes = /motorway|trunk/.test(step.roadClass || '') ? 2 : 1;
                    this.lanes = Array.from({length: n}, () => ({ dirs: ['straight'], active: false }));
                    for (let k = 0; k < this.rampLanes; k++) {
                        const i = rightSide ? n - 1 - k : k;
                        this.lanes[i] = { dirs: [this.maneuverDir === 'straight' ? (rightSide ? 'slight-right' : 'slight-left') : this.maneuverDir], active: true };
                    }
                    return;
                }

                // ── Default: build lanes with straight direction.
                this.lanes = Array.from({length: n}, () => ({ dirs: ['straight'], active: false }));
                const mid = Math.floor(n / 2);
                const md  = this.maneuverDir;

                // Sharp turn (>60°) → single dedicated lane, edge only.
                // Slight turn (<20°) → the through lane is also valid.
                const isSharp  = turnDeg >= 60;
                const isSlight = turnDeg > 0 && turnDeg < 20;

                if (md === 'uturn') {
                    this.lanes[0] = { dirs: ['uturn'], active: true };
                } else if (md === 'sharp-left' || (md === 'left' && isSharp)) {
                    this.lanes[0] = { dirs: ['left'], active: true };
                } else if (md === 'left' || md === 'slight-left') {
                    this.lanes[0] = { dirs: ['left'], active: true };
                    if (isSlight && n >= 3) this.lanes[1] = { dirs: ['slight-left'], active: true };
                } else if (md === 'sharp-right' || (md === 'right' && isSharp)) {
                    this.lanes[n - 1] = { dirs: ['right'], active: true };
                } else if (md === 'right' || md === 'slight-right') {
                    this.lanes[n - 1] = { dirs: ['right'], active: true };
                    if (isSlight && n >= 3) this.lanes[n - 2] = { dirs: ['slight-right'], active: true };
                } else if (md === 'arrive') {
                    // Prefer the side matching the destination (right in RHT countries)
                    this.lanes[n - 1] = { dirs: ['arrive'], active: true };
                } else {
                    // Straight: middle lane; on wide roads middle + through-right.
                    this.lanes[mid] = { dirs: ['straight'], active: true };
                    if (n >= 4) this.lanes[mid + 1] = { dirs: ['straight'], active: true };
                }
            }

            // Turn angle in degrees using bearings from OSRM when available,
            // otherwise derived from the first two step-geometry vertices vs
            // the previous step's tail direction.
            _estimateTurnDeg(step) {
                let before = step.bearingBefore, after = step.bearing;
                if (typeof before !== 'number' || typeof after !== 'number') {
                    // Fallback from modifier when no geometry to work with
                    if (!step.geometry || !step.geometry.coordinates ||
                        step.geometry.coordinates.length < 2) {
                        const md = this.maneuverDir;
                        if (md === 'sharp-left' || md === 'sharp-right' || md === 'uturn') return 90;
                        if (md === 'left' || md === 'right')                                return 55;
                        if (md === 'slight-left' || md === 'slight-right')                  return 15;
                        return 0;
                    }
                    const c = step.geometry.coordinates;
                    after = this._bearingRad(c[0], c[1]);
                    before = after; // no prev info; return 0
                    return 0;
                }
                let d = Math.abs(((after - before + 540) % 360) - 180);
                return d;
            }

            _bearingRad(a, b) {
                const φ1 = a[1] * Math.PI / 180, φ2 = b[1] * Math.PI / 180;
                const λ1 = a[0] * Math.PI / 180, λ2 = b[0] * Math.PI / 180;
                const y = Math.sin(λ2 - λ1) * Math.cos(φ2);
                const x = Math.cos(φ1) * Math.sin(φ2) - Math.sin(φ1) * Math.cos(φ2) * Math.cos(λ2 - λ1);
                return (Math.atan2(y, x) * 180 / Math.PI + 360) % 360;
            }

            // ── Loop ─────────────────────────────────────────────────────

            _startLoop() {
                // Throttle to ~40fps (25ms min). The scene is smooth enough
                // at 40 and this halves work on high-refresh displays.
                const minDt = 1 / 40;
                if (!this._visHooked) {
                    this._visHooked = true;
                    document.addEventListener('visibilitychange', () => {
                        // Reset lastTs so we don't inject a huge dt after resume
                        this.lastTs = 0;
                    });
                }
                const tick = (ts) => {
                    this.animFrame = requestAnimationFrame(tick);
                    if (document.hidden || this.hud.style.display === 'none') return;
                    const rawDt = this.lastTs ? (ts - this.lastTs) / 1000 : 0.016;
                    if (rawDt < minDt) return;               // FPS throttle
                    const dt = Math.min(0.05, rawDt);
                    this.lastTs = ts;
                    this.phase        += dt;
                    this.chevronPhase += dt;
                    const target = this._effectiveCurve();
                    this.curve += (target - this.curve) * Math.min(1, dt * 4.5);
                    this._draw();
                };
                this.animFrame = requestAnimationFrame(tick);
            }

            // Distance-based pre-turn easing: at 300m out, blend already
            // starts (0.15); at 30m, we're at full magnitude.
            _preTurnFactor() {
                const d = this.distanceMeters;
                if (!d) return 1;
                if (d <= 30)   return 1;
                if (d >= 300)  return 0.15;
                // smoothstep from 300 → 30
                const t = 1 - (d - 30) / 270;
                return 0.15 + (t * t * (3 - 2 * t)) * 0.85;
            }

            _effectiveCurve() {
                return this.curveBase * this._preTurnFactor();
            }

            // ── Projection ───────────────────────────────────────────────

            _horizonY() {
                // Bearing offset also shifts the horizon a little for a
                // "camera tilt / yaw" feel. Slight parallax only.
                return 30 + this.mapBearingOffset * 6;
            }

            _project(u, t) {
                const W = this.W, H = this.H;
                const vy = this._horizonY();
                const baseHalf = W * 0.62;
                const topHalf  = W * 0.13;
                const halfW = baseHalf + (topHalf - baseHalf) * t;
                const bend = this.curve * t * t * (W * 0.55);
                const yaw  = this.mapBearingOffset * 40; // pixel shift near cam
                const cx = W / 2 + bend + yaw * (1 - t);
                const x  = cx + (u - 0.5) * 2 * halfW;
                const y  = H + (vy - H) * t;
                return [x, y];
            }

            // ── Draw ─────────────────────────────────────────────────────

            _palette() {
                // Memoize — palette object shape doesn't change per frame,
                // only when nightMode flips.
                if (this._paletteCache && this._paletteNight === this.nightMode)
                    return this._paletteCache;
                this._paletteNight = this.nightMode;
                this._paletteCache = this.nightMode ? {
                    skyTop:'#01010a', skyMid:'#03030d', skyBot:'#08081a',
                    star: true, headlights: true,
                    chevron:'#ffb020', chevronGlow:'#ffb020',
                    edge:'rgba(255,240,180,0.72)',
                    edgeActive:'rgba(255,80,80,0.9)',
                    laneFillMid:'rgba(255,176,32,0.20)',
                    laneFillBot:'rgba(255,176,32,0.42)',
                    dividerCenter:'rgba(255,204,90,0.65)',
                    dividerSide:'rgba(230,235,255,0.42)',
                    signBg:'rgba(10,12,22,0.92)',
                    signText:'#8fd8ff',
                    signBorder:'rgba(140,200,255,0.55)',
                    fog:'rgba(140,180,255,0.10)',
                    streetlight:true
                } : {
                    skyTop:'#02020a', skyMid:'#06060f', skyBot:'#0e0e1c',
                    star: false, headlights:false,
                    chevron:'#ff5252', chevronGlow:'#ff3333',
                    edge:'rgba(240,245,255,0.75)',
                    edgeActive:'rgba(255,80,80,0.75)',
                    laneFillMid:'rgba(255,51,51,0.22)',
                    laneFillBot:'rgba(255,51,51,0.46)',
                    dividerCenter:'rgba(255,195,45,0.55)',
                    dividerSide:'rgba(230,235,255,0.45)',
                    signBg:'rgba(10,10,16,0.92)',
                    signText:'#ff8f8f',
                    signBorder:'rgba(255,80,80,0.55)',
                    fog:'rgba(255,51,51,0.11)',
                    streetlight:false
                };
                this._backdropBmp = null; // invalidate baked backdrop
                return this._paletteCache;
            }

            // Backdrop bakes the non-animated pixels (sky, stars, side
            // atmosphere, ground band below horizon) into an offscreen
            // canvas we blit each frame. Only rebuilds when palette / size
            // changes, cutting several gradient/fill ops per frame.
            _ensureBackdrop(P) {
                if (this._backdropBmp && this._backdropW === this.W && this._backdropH === this.H)
                    return this._backdropBmp;
                const W = this.W, H = this.H;
                const off = document.createElement('canvas');
                off.width = W * this.dpr; off.height = H * this.dpr;
                const octx = off.getContext('2d');
                octx.setTransform(this.dpr, 0, 0, this.dpr, 0, 0);

                const bg = octx.createLinearGradient(0, 0, 0, H);
                bg.addColorStop(0,    P.skyTop);
                bg.addColorStop(0.20, P.skyMid);
                bg.addColorStop(1,    P.skyBot);
                octx.fillStyle = bg; octx.fillRect(0, 0, W, H);

                if (P.star) {
                    for (let i = 0; i < 32; i++) {
                        const sx = (i * 53.7) % W;
                        const sy = 4 + ((i * 17) % 22);
                        octx.fillStyle = `rgba(200,220,255,${0.30 + ((i * 7) % 5) * 0.05})`;
                        octx.fillRect(sx, sy, 1, 1);
                    }
                }
                for (let side = 0; side <= 1; side++) {
                    const ag = octx.createRadialGradient(side * W, H, 0, side * W, H, W * 0.6);
                    ag.addColorStop(0, P.fog);
                    ag.addColorStop(1, 'rgba(0,0,0,0)');
                    octx.fillStyle = ag; octx.fillRect(0, 0, W, H);
                }
                // Ground band under a nominal horizon
                octx.fillStyle = '#05050f';
                octx.fillRect(0, 30, W, H - 30);

                this._backdropBmp = off;
                this._backdropW = W; this._backdropH = H;
                return off;
            }

            _draw() {
                const ctx = this.ctx;
                const W = this.W, H = this.H;
                const P = this._palette();

                // Blit the baked backdrop instead of rebuilding gradients.
                const bd = this._ensureBackdrop(P);
                ctx.drawImage(bd, 0, 0, W, H);

                this._fillRoad(P);
                if (this.isRamp) this._fillExitFork(P);
                this._drawDepthLines(P);
                this._drawActiveLaneFills(P);
                this._drawLaneDividers(P);
                this._drawEdge(0, P);
                this._drawEdge(1, P);
                if (P.streetlight) this._drawStreetlights(P);
                this._drawGroundChevrons(P);
                this._drawLaneChooser(P);
                if (this.maneuverType === 'arrive') this._drawBeacon(P);
                if (this.nextStreet || this.exitNumber) this._drawSignGantry(P);
                if (P.headlights) this._drawHeadlightCones();

                const vy = this._horizonY();
                const hg = ctx.createLinearGradient(0, vy - 14, 0, vy + 24);
                hg.addColorStop(0,    'rgba(0,0,0,0)');
                hg.addColorStop(0.45, P.fog);
                hg.addColorStop(1,    'rgba(0,0,0,0)');
                ctx.fillStyle = hg;
                ctx.fillRect(0, vy - 14, W, 38);

                this._drawSpeedStreaks(P);
                this._drawProgress(P);

                const vg = ctx.createRadialGradient(W / 2, H * 0.55, W * 0.25, W / 2, H * 0.55, W * 0.75);
                vg.addColorStop(0, 'rgba(0,0,0,0)');
                vg.addColorStop(1, 'rgba(0,0,0,0.55)');
                ctx.fillStyle = vg;
                ctx.fillRect(0, 0, W, H);
            }

            _roadPath(ctx) {
                const steps = 14;
                ctx.beginPath();
                for (let i = 0; i <= steps; i++) {
                    const t = i / steps;
                    const [x, y] = this._project(0, t);
                    if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y);
                }
                for (let i = steps; i >= 0; i--) {
                    const t = i / steps;
                    const [x, y] = this._project(1, t);
                    ctx.lineTo(x, y);
                }
                ctx.closePath();
            }

            _fillRoad(P) {
                const ctx = this.ctx;
                this._roadPath(ctx);
                if (!this._roadGrad || this._roadGradNight !== this.nightMode) {
                    this._roadGradNight = this.nightMode;
                    const g = ctx.createLinearGradient(0, 30, 0, this.H);
                    if (this.nightMode) {
                        g.addColorStop(0, '#08080f'); g.addColorStop(0.55, '#0d0d18'); g.addColorStop(1, '#12121e');
                    } else {
                        g.addColorStop(0, '#0c0c1b'); g.addColorStop(0.55, '#131324'); g.addColorStop(1, '#1b1b2a');
                    }
                    this._roadGrad = g;
                }
                ctx.fillStyle = this._roadGrad;
                ctx.fill();
            }

            // A gently diverging branch that peels off toward the maneuver
            // side, meeting the road at the vanishing point.
            _fillExitFork(P) {
                const ctx = this.ctx;
                const rightSide = !/left/.test(this.maneuverDir);
                const steps = 14;
                ctx.save();
                ctx.beginPath();
                for (let i = 0; i <= steps; i++) {
                    const t = i / steps;
                    // Inner edge starts at the outermost lane divider
                    const u = rightSide ? 1 - (1 / this.numLanes) : (1 / this.numLanes);
                    const [x, y] = this._project(u, t);
                    if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y);
                }
                for (let i = steps; i >= 0; i--) {
                    const t = i / steps;
                    // Outer edge sweeps away from the main road
                    const drift = rightSide ? 0.45 * t : -0.45 * t;
                    const u = rightSide ? 1 + drift : 0 + drift;
                    const [x, y] = this._project(u, t);
                    ctx.lineTo(x, y);
                }
                ctx.closePath();
                const g = ctx.createLinearGradient(0, 30, 0, this.H);
                g.addColorStop(0,    'rgba(255,80,80,0.10)');
                g.addColorStop(0.55, 'rgba(255,80,80,0.22)');
                g.addColorStop(1,    'rgba(255,80,80,0.42)');
                ctx.fillStyle = g;
                ctx.fill();

                // Fork divider gore (V-shape)
                ctx.beginPath();
                const [gx, gy] = this._project(rightSide ? 1 - 1 / this.numLanes : 1 / this.numLanes, 0.02);
                const [vx, vy] = this._project(rightSide ? 1 - 1 / this.numLanes : 1 / this.numLanes, 0.85);
                ctx.moveTo(gx, gy);
                ctx.lineTo(vx, vy);
                ctx.strokeStyle = P.chevron;
                ctx.lineWidth = 1.6;
                ctx.setLineDash([4, 3]);
                ctx.stroke();
                ctx.setLineDash([]);
                ctx.restore();
            }

            _drawDepthLines(P) {
                const ctx = this.ctx;
                [0.20, 0.34, 0.48, 0.62, 0.75, 0.86, 0.94].forEach(t => {
                    const [xL, yL] = this._project(0, t);
                    const [xR, yR] = this._project(1, t);
                    ctx.beginPath();
                    ctx.moveTo(xL, yL); ctx.lineTo(xR, yR);
                    ctx.strokeStyle = `rgba(180,190,220,${0.04 + (1 - t) * 0.08})`;
                    ctx.lineWidth = 0.5;
                    ctx.stroke();
                });
            }

            _laneEdgeU(i) { return i / this.numLanes; }

            _drawActiveLaneFills(P) {
                const ctx = this.ctx;
                const pulse = 0.5 + 0.25 * Math.sin(this.phase * 2.6);
                this.lanes.forEach((lane, i) => {
                    const isActive = lane.active;
                    const isUser   = this.userPickedLane === i;
                    if (!isActive && !isUser && this.hoverLane !== i) return;

                    const uL = this._laneEdgeU(i), uR = this._laneEdgeU(i + 1);
                    ctx.beginPath();
                    const steps = 12;
                    for (let s = 0; s <= steps; s++) {
                        const t = s / steps;
                        const [x, y] = this._project(uL, t);
                        if (s === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y);
                    }
                    for (let s = steps; s >= 0; s--) {
                        const t = s / steps;
                        const [x, y] = this._project(uR, t);
                        ctx.lineTo(x, y);
                    }
                    ctx.closePath();
                    const g = ctx.createLinearGradient(0, 30, 0, this.H);
                    if (isUser) {
                        g.addColorStop(0,    'rgba(80,220,140,0.05)');
                        g.addColorStop(0.55, `rgba(80,220,140,${0.32 * pulse + 0.1})`);
                        g.addColorStop(1,    `rgba(80,220,140,${0.58 * pulse + 0.1})`);
                    } else if (isActive) {
                        g.addColorStop(0,    'rgba(255,51,51,0.02)');
                        g.addColorStop(0.55, P.laneFillMid);
                        g.addColorStop(1,    P.laneFillBot);
                    } else {
                        // hover
                        g.addColorStop(0,    'rgba(255,255,255,0.02)');
                        g.addColorStop(0.55, 'rgba(255,255,255,0.06)');
                        g.addColorStop(1,    'rgba(255,255,255,0.10)');
                    }
                    ctx.fillStyle = g;
                    ctx.fill();
                });
            }

            _drawLaneDividers(P) {
                const ctx = this.ctx;
                const dashWorld = 0.06;
                const gapWorld  = 0.05;
                const scroll    = (this.phase * 0.55) % (dashWorld + gapWorld);

                for (let i = 1; i < this.numLanes; i++) {
                    const u = this._laneEdgeU(i);
                    const isCenter = (this.numLanes % 2 === 1) && (i === Math.floor(this.numLanes / 2));
                    ctx.strokeStyle = isCenter ? P.dividerCenter : P.dividerSide;
                    ctx.lineWidth = 1.4;

                    let t = -scroll;
                    while (t < 1) {
                        const t0 = Math.max(0, t);
                        const t1 = Math.min(1, t + dashWorld);
                        if (t1 > t0) {
                            const [x0, y0] = this._project(u, t0);
                            const [x1, y1] = this._project(u, t1);
                            ctx.beginPath();
                            ctx.moveTo(x0, y0);
                            ctx.lineTo(x1, y1);
                            ctx.stroke();
                        }
                        t += dashWorld + gapWorld;
                    }
                }
            }

            _drawEdge(side, P) {
                const ctx = this.ctx;
                const u = side;
                const steps = 16;
                ctx.beginPath();
                for (let s = 0; s <= steps; s++) {
                    const t = s / steps;
                    const [x, y] = this._project(u, t);
                    if (s === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y);
                }
                ctx.strokeStyle = P.edge;
                ctx.lineWidth = 1.8;
                ctx.stroke();

                const activeAtEdge = this.lanes.some((l, i) =>
                    l.active && (side === 0 ? i === 0 : i === this.numLanes - 1));
                if (activeAtEdge) {
                    ctx.save();
                    ctx.shadowColor = P.chevronGlow;
                    ctx.shadowBlur  = 8;
                    ctx.strokeStyle = P.edgeActive;
                    ctx.lineWidth = 1.4;
                    ctx.stroke();
                    ctx.restore();
                }
            }

            _drawStreetlights(P) {
                const ctx = this.ctx;
                for (let side = 0; side <= 1; side++) {
                    for (let k = 0; k < 4; k++) {
                        const t = 0.15 + k * 0.22 + (this.phase * 0.05) % 0.22;
                        if (t > 0.9) continue;
                        const u = side === 0 ? -0.05 : 1.05;
                        const [x, y] = this._project(u, t);
                        const size = (1 - t) * 2.4 + 0.6;
                        ctx.save();
                        ctx.shadowColor = '#ffe08a';
                        ctx.shadowBlur = 8;
                        ctx.fillStyle = '#ffe08a';
                        ctx.beginPath();
                        ctx.arc(x, y - size * 2, size, 0, Math.PI * 2);
                        ctx.fill();
                        ctx.restore();
                    }
                }
            }

            _drawGroundChevrons(P) {
                const ctx = this.ctx;
                const speed = 0.35;
                const spacing = 0.22;
                const base = (this.chevronPhase * speed) % spacing;

                this.lanes.forEach((lane, i) => {
                    if (!lane.active) return;
                    const uC = (i + 0.5) / this.numLanes;
                    const dir = (lane.dirs && lane.dirs[0]) || this.maneuverDir;

                    for (let k = 0; k < 5; k++) {
                        const t = 0.05 + base + k * spacing;
                        if (t > 0.92) continue;
                        const localDir = this._chevronDirAt(dir, t);
                        const [cx, cy] = this._project(uC, t);
                        const [nx, ny] = this._project(uC, Math.min(0.99, t + 0.02));
                        const ang = Math.atan2(ny - cy, nx - cx) - Math.PI / 2;
                        const size = (1 - t) * 18 + 4;
                        const opacity = Math.min(1, (1 - t) * 1.4);
                        this._drawGroundArrow(ctx, cx, cy, size, ang, localDir, opacity, P.chevron, P.chevronGlow);
                    }
                });

                // Ghost chevrons for the NEXT maneuver, blended in near horizon
                if (this.nextManeuver) {
                    const uC = 0.5;
                    for (let k = 0; k < 2; k++) {
                        const t = 0.72 + k * 0.10;
                        const [cx, cy] = this._project(uC, t);
                        const [nx, ny] = this._project(uC, Math.min(0.99, t + 0.02));
                        const ang = Math.atan2(ny - cy, nx - cx) - Math.PI / 2;
                        const size = (1 - t) * 18 + 4;
                        const opacity = Math.max(0, (1 - t) * 0.9) * 0.55;
                        this._drawGroundArrow(ctx, cx, cy, size, ang, this.nextManeuver.dir,
                                              opacity, '#8fd8ff', '#8fd8ff');
                    }
                }
            }

            _chevronDirAt(dir, t) {
                // Now curves earlier (t > 0.35) so the "into the turn" feel
                // starts before the maneuver point — matches early bend.
                if (t < 0.35) return 'straight';
                return dir;
            }

            _drawGroundArrow(ctx, x, y, s, ang, dir, alpha, fill, glow) {
                ctx.save();
                ctx.translate(x, y);
                ctx.rotate(ang);
                ctx.scale(1, 0.55);

                ctx.globalAlpha = alpha;
                ctx.strokeStyle = fill;
                ctx.fillStyle   = fill;
                ctx.lineWidth   = Math.max(1.2, s * 0.18);
                ctx.lineCap = 'round';
                ctx.lineJoin = 'round';
                ctx.shadowColor = glow;
                ctx.shadowBlur  = 6;

                if (dir === 'left' || dir === 'slight-left' || dir === 'sharp-left') {
                    ctx.beginPath();
                    ctx.moveTo(0,  s * 0.55);
                    ctx.lineTo(0,  0);
                    ctx.quadraticCurveTo(0, -s * 0.55, -s * 0.55, -s * 0.55);
                    ctx.stroke();
                    ctx.beginPath();
                    ctx.moveTo(-s * 0.90, -s * 0.55);
                    ctx.lineTo(-s * 0.35, -s * 0.25);
                    ctx.lineTo(-s * 0.35, -s * 0.85);
                    ctx.closePath(); ctx.fill();
                } else if (dir === 'right' || dir === 'slight-right' || dir === 'sharp-right') {
                    ctx.beginPath();
                    ctx.moveTo(0,  s * 0.55);
                    ctx.lineTo(0,  0);
                    ctx.quadraticCurveTo(0, -s * 0.55, s * 0.55, -s * 0.55);
                    ctx.stroke();
                    ctx.beginPath();
                    ctx.moveTo(s * 0.90, -s * 0.55);
                    ctx.lineTo(s * 0.35, -s * 0.25);
                    ctx.lineTo(s * 0.35, -s * 0.85);
                    ctx.closePath(); ctx.fill();
                } else if (dir === 'uturn') {
                    ctx.beginPath();
                    ctx.moveTo(s * 0.30,  s * 0.55);
                    ctx.lineTo(s * 0.30, -s * 0.10);
                    ctx.arc(0, -s * 0.10, s * 0.30, 0, Math.PI, true);
                    ctx.lineTo(-s * 0.30, s * 0.30);
                    ctx.stroke();
                    ctx.beginPath();
                    ctx.moveTo(-s * 0.65, s * 0.15);
                    ctx.lineTo(-s * 0.30, s * 0.00);
                    ctx.lineTo(-s * 0.05, s * 0.15);
                    ctx.closePath(); ctx.fill();
                } else if (dir === 'arrive') {
                    ctx.beginPath();
                    ctx.arc(0, 0, s * 0.55, 0, Math.PI * 2);
                    ctx.fill();
                } else {
                    ctx.beginPath();
                    ctx.moveTo(-s * 0.65, s * 0.10);
                    ctx.lineTo(0,        -s * 0.55);
                    ctx.lineTo( s * 0.65, s * 0.10);
                    ctx.stroke();
                    ctx.beginPath();
                    ctx.moveTo(-s * 0.65, s * 0.55);
                    ctx.lineTo(0,        -s * 0.05);
                    ctx.lineTo( s * 0.65, s * 0.55);
                    ctx.stroke();
                }
                ctx.restore();
            }

        // ── Lane-guidance ⇄ map synchronization ─────────────────────────
        // Keeps the 3D HUD's horizon/curve aligned with wherever the user
        // has rotated the map, and updates the pre-turn distance blend as
        // GPS progresses toward the next maneuver.
        let activeStepIndex = 0;
        function currentActiveStep() {
            const r = activeRouteDataStore[selectedRouteIndex];
            if (!r || !r.steps.length) return null;
            return r.steps[Math.min(activeStepIndex, r.steps.length - 1)];
        }
        function pushMapCameraDelta() {
            if (!laneGuidance || !map) return;
            const step = currentActiveStep();
            if (!step || typeof step.bearing !== 'number') {
                laneGuidance.setMapCameraDelta(0);
                return;
            }
            // Signed shortest-angle delta between map view and turn bearing
            let d = ((map.getBearing() - step.bearing + 540) % 360) - 180;
            laneGuidance.setMapCameraDelta((d * Math.PI) / 180);
        }
        function wireLaneGuidanceMapSync() {
            map.on('rotate', pushMapCameraDelta);
            map.on('pitch',  pushMapCameraDelta);
            map.on('move',   pushMapCameraDelta);
        }

        // Approx meters between two [lng,lat] points (equirectangular)
        function metersBetween(a, b) {
            const R = 6371000;
            const lat1 = a[1] * Math.PI / 180, lat2 = b[1] * Math.PI / 180;
            const x = (b[0] - a[0]) * Math.PI / 180 * Math.cos((lat1 + lat2) / 2);
            const y = (b[1] - a[1]) * Math.PI / 180;
            return Math.sqrt(x * x + y * y) * R;
        }

        // Called from the GPS loop with the latest fix [lng,lat]
        function syncLaneGuidanceLive(fix) {
            if (!laneGuidance) return;
            const r = activeRouteDataStore[selectedRouteIndex];
            if (!r || !r.steps.length || !fix) return;

            // Advance the active step when we're within 20m of its maneuver
            const cur = r.steps[activeStepIndex];
            if (cur && cur.coordinates) {
                const d = metersBetween(fix, cur.coordinates);
                if (d < 20 && activeStepIndex < r.steps.length - 1) {
                    activeStepIndex++;
                    const s = r.steps[activeStepIndex];
                    laneGuidance.show(s, r.steps[activeStepIndex + 1] || null);
                    updateMapLaneGuidance();
                } else {
                    laneGuidance.updateDistance(d);
                }
            }
            pushMapCameraDelta();
        }

        // ── On-map lane guidance ─────────────────────────────────────────
        // Draws the upcoming maneuver's recommended lane path and a chain
        // of directional arrows directly onto the MapLibre map, oriented
        // to the actual road geometry.
        function ensureLaneArrowIcon() {
            if (!map || map.hasImage('lane-arrow')) return;
            const s = 48;
            const c = document.createElement('canvas'); c.width = s; c.height = s;
            const g = c.getContext('2d');
            // Arrow pointing UP (symbol layer rotates it along the line)
            g.fillStyle = '#ffffff';
            g.beginPath();
            g.moveTo(s / 2, 4);
            g.lineTo(s - 8, s / 2 + 2);
            g.lineTo(s / 2 + 7, s / 2 + 2);
            g.lineTo(s / 2 + 7, s - 6);
            g.lineTo(s / 2 - 7, s - 6);
            g.lineTo(s / 2 - 7, s / 2 + 2);
            g.lineTo(8,         s / 2 + 2);
            g.closePath();
            g.fill();
            // Slight outline for contrast on light basemaps
            g.strokeStyle = 'rgba(0,0,0,0.35)'; g.lineWidth = 1.5; g.stroke();
            const img = g.getImageData(0, 0, s, s);
            map.addImage('lane-arrow', img, { sdf: true, pixelRatio: 2 });
        }

        // Rough px offset from centerline into the recommended lane at
        // typical navigation zoom levels. Uses lane index shift from center.
        function laneOffsetPx(laneIdx, numLanes) {
            const shift = (laneIdx + 0.5) - numLanes / 2;
            return shift * 5; // ≈5 px per lane at z17 — MapLibre offsets in screen px
        }

        function updateMapLaneGuidance() {
            if (!map || !map.getSource || !map.getSource('lane-guidance')) return;
            const empty = { type: 'FeatureCollection', features: [] };
            const r = activeRouteDataStore[selectedRouteIndex];
            if (!r || !r.steps.length || !laneGuidance) {
                map.getSource('lane-guidance').setData(empty);
                return;
            }
            const step = r.steps[Math.min(activeStepIndex, r.steps.length - 1)];
            if (!step || !step.geometry || !Array.isArray(step.geometry.coordinates) ||
                step.geometry.coordinates.length < 2) {
                map.getSource('lane-guidance').setData(empty);
                return;
            }

            const night = laneGuidance.nightMode;
            const activeIdx = (laneGuidance.userPickedLane != null)
                ? laneGuidance.userPickedLane
                : (laneGuidance.lanes.findIndex(l => l.active));
            const idx = activeIdx >= 0 ? activeIdx : Math.floor(laneGuidance.numLanes / 2);
            const offset = laneOffsetPx(idx, laneGuidance.numLanes);

            const userPicked = laneGuidance.userPickedLane != null;
            const color = userPicked ? '#3fd18a'
                        : night      ? '#ffb020'
                                     : '#ff5252';
            const halo  = night ? '#221100' : '#3a0000';

            const feature = {
                type: 'Feature',
                geometry: step.geometry,
                properties: { color, halo, offset }
            };
            map.getSource('lane-guidance').setData({ type: 'FeatureCollection', features: [feature] });
        }

        async function handleRoutingButtonClick() {
            const btn = document.getElementById('launch-routing-action');
            if (!btn || !isValidCoordinatePair(targetSelectedCoordinates)) return;

            btn.innerText = "COMPUTING PATH TRAJECTORY...";
            btn.style.opacity = "0.6";
            btn.disabled = true;

            let originCoords = isValidCoordinatePair(currentLiveGpsPosition) ? currentLiveGpsPosition : null;
            if (!originCoords) {
                const mapCenter = map.getCenter();
                originCoords = [mapCenter.lng, mapCenter.lat];
            }
            if (!isValidCoordinatePair(originCoords)) {
                btn.innerText = "GPS ORIGIN UNAVAILABLE";
                btn.style.opacity = "1";
                btn.disabled = false;
                return;
            }

            const routingQueryUrl = `https://router.project-osrm.org/route/v1/${currentRoutingProfile}/${originCoords.join(',')};${targetSelectedCoordinates.join(',')}?overview=full&geometries=geojson&steps=true&alternatives=3`;

            try {
                clearActiveRouteLayers();
                activeRouteDataStore = [];
                activeStepIndex = 0;

                const stream = await fetch(routingQueryUrl);
                if (!stream.ok) throw new Error(`Routing service returned ${stream.status}`);
                const results = await stream.json();

                if (results.code === 'Ok' && results.routes && results.routes.length > 0) {
                    activeRouteDataStore = results.routes.slice(0, 3).map((route, idx) => {
                        let parsedInstructions = [];
                        if (route.legs && route.legs[0].steps) {
                            const rawSteps = route.legs[0].steps;
                            parsedInstructions = rawSteps.map((step, si) => {
                                const nextRaw = rawSteps[si + 1];
                                let modifier = step.maneuver.modifier ? ` ${step.maneuver.modifier}` : '';
                                let type = step.maneuver.type === 'turn' ? 'Turn' : step.maneuver.type;
                                let street = step.name ? ` onto ${step.name}` : '';
                                let rawMessage = `${type}${modifier}${street}`;
                                if (step.maneuver.type === 'depart') rawMessage = `Follow target vector from ${step.name || 'current coordinates'}`;
                                if (step.maneuver.type === 'arrive') rawMessage = "Target location node reached";

                                // Exit number (OSRM/Mapbox conventions)
                                const exitNum = (step.maneuver && step.maneuver.exit)
                                    || (step.exits ? String(step.exits).split(';')[0] : '')
                                    || (step.destinations && /Exit\s+(\S+)/i.test(step.destinations)
                                          ? step.destinations.match(/Exit\s+(\S+)/i)[1] : '');

                                // Lane hints from the first intersection that has them
                                let lanes = null;
                                if (Array.isArray(step.intersections)) {
                                    const withLanes = step.intersections.find(x => Array.isArray(x.lanes) && x.lanes.length);
                                    if (withLanes) lanes = withLanes.lanes;
                                }

                                // Target compass bearing for this step (for map sync)
                                let bearing = null, bearingBefore = null;
                                if (Array.isArray(step.intersections) && step.intersections[0]) {
                                    const inter = step.intersections[0];
                                    if (typeof inter.bearing_after === 'number')       bearing = inter.bearing_after;
                                    else if (Array.isArray(inter.bearings) && Array.isArray(inter.entry)) {
                                        const idxB = inter.entry.findIndex(v => v === true);
                                        if (idxB >= 0) bearing = inter.bearings[idxB];
                                    }
                                    if (typeof inter.bearing_before === 'number') bearingBefore = inter.bearing_before;
                                }

                                // Road class heuristic from OSRM step
                                const roadClass = (step.mode === 'driving' && step.ref) ? 'primary' :
                                                  (step.name || '').toLowerCase().match(/\b(hwy|highway|fwy|freeway|expressway|interstate|i-\d)\b/) ? 'motorway' :
                                                  (step.ref ? 'trunk' : 'street');

                                return {
                                    text: rawMessage.charAt(0).toUpperCase() + rawMessage.slice(1),
                                    distance: step.distance,
                                    coordinates: step.maneuver.location,
                                    maneuverType: step.maneuver.type,
                                    maneuverModifier: step.maneuver.modifier || 'straight',
                                    name: step.name || '',
                                    nextName: nextRaw ? (nextRaw.name || '') : '',
                                    exit: exitNum || '',
                                    lanes: lanes,
                                    bearing: bearing,
                                    bearingBefore: bearingBefore,
                                    geometry: step.geometry || null,
                                    roadClass: roadClass,
                                    ref: step.ref || '',
                                    destinations: step.destinations || ''
                                };
                            }).filter(item => item.text);
                        }
                        return { index: idx, geometry: route.geometry, duration: route.duration, distance: route.distance, steps: parsedInstructions };
                    });
                    selectedRouteIndex = 0;
                }
            } catch (err) {
                console.error("OSRM route fault.", err);
            }

            redrawActiveRoutePipelines();
            renderRouteDashboardUI();

            if (activeRouteDataStore.length > 0) {
                const coords = activeRouteDataStore[0].geometry.coordinates;
                if (coords.length > 1) {
                    const initialAngle = calculateBearingAngle(coords[0], coords[1]);
                    map.flyTo({
                        center: coords[0],
                        zoom: 17.5,
                        pitch: 70,
                        bearing: initialAngle,
                        duration: 1200
                    });
                }
                document.getElementById('control-panel-deck').classList.add('minimized');
                // Show 3D lane guidance starting from the first step of selected route
                if (laneGuidance && activeRouteDataStore[selectedRouteIndex].steps.length > 0) {
                    const s = activeRouteDataStore[selectedRouteIndex].steps;
                    laneGuidance.show(s[0], s[1] || null);
                    updateMapLaneGuidance();
                }
                // Load elevation profile
                loadElevationProfile(activeRouteDataStore[selectedRouteIndex].geometry.coordinates);
            } else {
                btn.innerText = "ROUTE UNAVAILABLE - TRY AGAIN";
                btn.style.display = "block";
            }

            btn.style.opacity = "1";
            btn.disabled = false;

            // FIX #4: Reset button text before hiding so it shows correctly
            // if the panel is re-expanded before the next destination is selected.
            if (activeRouteDataStore.length > 0) {
                btn.innerText = "GET DIRECTIONS";
                btn.style.display = "none";
            }
        }

                    </div>
                </div>
            `;

        function establishLiveHardwarePositioningStream() {
            const diagnosticStatus = document.getElementById('telemetry-status-node');
            if (!diagnosticStatus) return;

            if (!window.isSecureContext) {
                diagnosticStatus.innerText = "GPS REQUIRES HTTPS";
                diagnosticStatus.style.color = "#ff9500";
                return;
            }

            if (!navigator.geolocation) {
                diagnosticStatus.innerText = "SENSOR FAULT";
                diagnosticStatus.style.color = "#ff9500";
                return;
            }

            let initialLockAcquired = false;

            navigator.geolocation.watchPosition(
                (position) => {
                    const rawFix = [position.coords.longitude, position.coords.latitude];
                    if (!isValidCoordinatePair(rawFix)) return;

                    const now = performance.now();
                    const accuracy = Number.isFinite(position.coords.accuracy) ? position.coords.accuracy : Infinity;
                    const hasExistingFix = isValidCoordinatePair(currentLiveGpsPosition);

                    if (hasExistingFix && accuracy > GPS_MAX_ACCEPTED_ACCURACY_METERS) {
                        diagnosticStatus.innerText = `GPS WEAK (${Math.round(accuracy)}M)`;
                        diagnosticStatus.style.color = "#ff9500";
                        return;
                    }

                    const elapsedSecs = lastGpsTimestamp ? Math.max((now - lastGpsTimestamp) / 1000, 0.001) : 0;
                    const movementMeters = hasExistingFix ? calculateDistanceMeters(currentLiveGpsPosition, rawFix) : 0;
                    const impliedSpeed = elapsedSecs > 0 ? movementMeters / elapsedSecs : 0;

                    if (hasExistingFix && elapsedSecs < 10 && impliedSpeed > GPS_MAX_PLAUSIBLE_SPEED_MPS) {
                        diagnosticStatus.innerText = "GPS SPIKE FILTERED";
                        diagnosticStatus.style.color = "#ff9500";
                        return;
                    }

                    const stableFix = initialLockAcquired
                        ? smoothCoordinatePair(currentLiveGpsPosition, rawFix)
                        : rawFix;

                    let nextHeading = currentVehicleHeading;
                    if (Number.isFinite(position.coords.heading) && position.coords.heading >= 0) {
                        nextHeading = position.coords.heading;
                    } else if (movementMeters >= GPS_MIN_HEADING_DISTANCE_METERS) {
                        nextHeading = calculateBearingAngle(currentLiveGpsPosition, rawFix, currentVehicleHeading);
                    }

                    currentLiveGpsPosition = stableFix;
                    currentVehicleHeading = nextHeading;
                    lastGpsTimestamp = now;

                    // Throttled weather + speed limit side effects
                    updateWeatherHUD(stableFix[1], stableFix[0]);
                    if (activeRouteDataStore.length > 0) updateSpeedLimit(stableFix[1], stableFix[0]);
                    syncLaneGuidanceLive(stableFix);

                    diagnosticStatus.innerText = accuracy === Infinity
                        ? "GPS TRACKING ACTIVE"
                        : `GPS TRACKING ACTIVE ±${Math.round(accuracy)}M`;
                    diagnosticStatus.style.color = "var(--action-red)";

                    if (!userLocationBeaconMarker) {
                        const beaconEl = document.createElement('div');
                        beaconEl.className = 'live-gps-beacon-node';
                        userLocationBeaconMarker = new maplibregl.Marker({ element: beaconEl })
                            .setLngLat(currentLiveGpsPosition)
                            .addTo(map);
                    } else {
                        userLocationBeaconMarker.setLngLat(currentLiveGpsPosition);
                    }

                    if (!initialLockAcquired) {
                        initialLockAcquired = true;
                        lastCameraUpdateTimestamp = now;
                        map.jumpTo({ center: currentLiveGpsPosition, zoom: 17.5, pitch: 70, bearing: currentVehicleHeading });
                        return;
                    }

                    if (cameraTrackingMode && now - lastCameraUpdateTimestamp >= GPS_MIN_CAMERA_INTERVAL_MS) {
                        lastCameraUpdateTimestamp = now;
                        map.easeTo({
                            center: currentLiveGpsPosition,
                            pitch: 70,
                            bearing: currentVehicleHeading,
                            zoom: Math.max(map.getZoom(), 17.5),
                            duration: 650,
                            essential: true
                        });
                    }
                },
                (error) => {
                    const messageByCode = {
                        [error.PERMISSION_DENIED]: "GPS PERMISSION DENIED",
                        [error.POSITION_UNAVAILABLE]: "GPS POSITION UNAVAILABLE",
                        [error.TIMEOUT]: "GPS TIMEOUT - RETRYING"
                    };
                    diagnosticStatus.innerText = messageByCode[error.code] || "GPS SENSOR ERROR";
                    diagnosticStatus.style.color = "#ff9500";
                },
                { enableHighAccuracy: true, timeout: 10000, maximumAge: 1000 }
            );
        }

        function initTacticalPanelDraggingPhysics() {
            const panel = document.getElementById('control-panel-deck');
            const handle = document.getElementById('drag-handle');
            if (!panel || !handle) return;

            // FIX #1: Use dragOccurred to distinguish real drags from clicks.
            // The bug was that isDragging is reset to false in endDrag (which
            // fires on mouseup) BEFORE the 'click' event fires. This meant the
            // !isDragging check in the click handler was always true, causing the
            // panel to both snap via endDrag AND toggle again via click on every drag.
            let isDragging = false, startY = 0, dragOccurred = false;

            handle.addEventListener('mousedown', startDrag);
            window.addEventListener('mousemove', processDragMouse);
            window.addEventListener('mouseup', endDrag);
            handle.addEventListener('touchstart', (e) => startDrag(e.touches[0]), { passive: true });

            // FIX #2: Call e.preventDefault() during touch drag so the page does
            // not scroll while the user is dragging the panel on mobile.
            // The listener was already registered as passive:false (enabling
            // preventDefault) but never actually called it.
            window.addEventListener('touchmove', (e) => {
                if (isDragging) e.preventDefault();
                processDragTouch(e.touches[0]);
            }, { passive: false });

            window.addEventListener('touchend', endDrag);

            handle.addEventListener('click', () => {
                // Only toggle if the pointer didn't meaningfully move (i.e. it
                // was a tap/click, not the end of a drag gesture).
                if (!dragOccurred) {
                    panel.classList.toggle('minimized');
                    panel.style.transform = '';
                }
                dragOccurred = false;
            });

            function startDrag(e) {
                isDragging = true;
                dragOccurred = false;
                startY = e.clientY;
                panel.classList.add('dragging');
            }

            function processDragMouse(e) {
                if (!isDragging) return;
                applyDrag(e.clientY);
            }

            function processDragTouch(e) {
                if (!isDragging) return;
                applyDrag(e.clientY);
            }

            function applyDrag(clientY) {
                const deltaY = clientY - startY;
                // Only count as a drag if the pointer moved more than 5px.
                if (Math.abs(deltaY) > 5) dragOccurred = true;

                if (panel.classList.contains('minimized')) {
                    if (deltaY < 0) {
                        panel.style.transform = `translateX(-50%) translateY(calc(100% - 28px + ${deltaY}px))`;
                    }
                } else {
                    if (deltaY > 0) {
                        panel.style.transform = `translateX(-50%) translateY(${deltaY}px)`;
                    }
                }
            }

            function endDrag(e) {
                if (!isDragging) return;
                isDragging = false;
                panel.classList.remove('dragging');
                panel.style.transform = '';

                const clientY = e.changedTouches ? e.changedTouches[0].clientY : e.clientY;
                const deltaY = clientY - startY;

                if (panel.classList.contains('minimized')) {
                    if (deltaY < -50) panel.classList.remove('minimized');
                } else {
                    if (deltaY > 50) panel.classList.add('minimized');
                }
            }
        }

        function initAutonomousSearchMatrix() {
            const input = document.getElementById('osm-engine-input');
            const dropdown = document.getElementById('osm-engine-dropdown');
            if (!input || !dropdown) return;

            input.addEventListener('input', (e) => {
                const queryVal = e.target.value.trim();
                clearTimeout(querySearchDebounceTimer);
                if (queryVal.length < 3) { dropdown.style.display = 'none'; dropdown.innerHTML = ''; return; }

                querySearchDebounceTimer = setTimeout(async () => {
                    try {
                        let searchApiUrl = `https://photon.komoot.io/api/?q=${encodeURIComponent(queryVal)}&limit=5`;
                        if (isValidCoordinatePair(currentLiveGpsPosition)) searchApiUrl += `&lon=${currentLiveGpsPosition[0]}&lat=${currentLiveGpsPosition[1]}`;

                        const response = await fetch(searchApiUrl);
                        if (!response.ok) throw new Error(`Search service returned ${response.status}`);
                        const payload = await response.json();

                        if (!payload.features || payload.features.length === 0) { dropdown.style.display = 'none'; return; }

                        dropdown.innerHTML = '';
                        payload.features.forEach(feature => {
                            const segmentItem = document.createElement('div');
                            segmentItem.className = 'osm-suggest-item';
                            const p = feature.properties;
                            const labeledLocation = [p.name, p.city, p.state, p.country].filter(Boolean).join(', ') || 'Unnamed location';
                            segmentItem.innerText = labeledLocation;

                            segmentItem.addEventListener('click', () => {
                                const targetLoc = feature.geometry.coordinates;
                                if (!isValidCoordinatePair(targetLoc)) return;

                                dropdown.style.display = 'none';
                                input.value = labeledLocation;

                                cameraTrackingMode = false;

        async function runPOISearch() {
            const center = isValidCoordinatePair(currentLiveGpsPosition) ? currentLiveGpsPosition : (() => { const c = map.getCenter(); return [c.lng, c.lat]; })();
            const list = document.getElementById('poi-results-list');
            if (list) list.innerHTML = '<div class="poi-status-msg">Scanning area…</div>';
            try {
                const cat = POI_CATS[_poiCat];
                const body = `[out:json][timeout:12];(${cat.q}(around:${_poiRadius},${center[1].toFixed(5)},${center[0].toFixed(5)}););out 20;`;
                const res = await fetch('https://overpass-api.de/api/interpreter', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/x-www-form-urlencoded; charset=UTF-8' },
                    body: `data=${encodeURIComponent(body)}`
                });
                if (!res.ok) throw new Error(`Overpass ${res.status}`);
                const data = await res.json();
                const pois = (data.elements || []).map(el => ({
                    name: el.tags?.name || el.tags?.brand || cat.lbl,
                    lat: el.lat, lon: el.lon,
                    dist: calculateDistanceMeters([center[0], center[1]], [el.lon, el.lat])
                })).sort((a,b) => a.dist - b.dist);

                if (!list) return;
                list.innerHTML = '';
                if (!pois.length) { list.innerHTML = `<div class="poi-status-msg">No ${cat.lbl} found nearby.<br>Try a larger radius.</div>`; return; }
                pois.slice(0, 18).forEach(p => {
                    const ft = Math.round(p.dist * 3.28084);
                    const distStr = ft >= 5280 ? `${(ft/5280).toFixed(1)} mi` : `${ft} ft`;
                    const card = document.createElement('button');
                    card.className = 'poi-card';
                    card.innerHTML = `<span class="poi-card-ico">${cat.ico}</span><span class="poi-card-body"><div class="poi-card-name">${p.name}</div><div class="poi-card-meta">${distStr} away</div></span><span class="poi-card-nav">GO →</span>`;
                    card.addEventListener('click', () => {
                        const coords = [p.lon, p.lat];
                        map.flyTo({ center: coords, zoom: 18, pitch: 65, duration: 800 });
                        triggerWorkspaceTargetDeck(coords, p.name, `${distStr} away · ${cat.lbl}`);
                        document.getElementById('poi-explorer-panel').style.display = 'none';
                        document.getElementById('poi-radar-toggle').classList.remove('active-state');
                    });
                    list.appendChild(card);
                });
            } catch(e) {
                if (list) list.innerHTML = '<div class="poi-status-msg">Search failed. Check connection.</div>';
                console.warn('POI fetch failed:', e.message);
            }
        }

        // ── ELEVATION PROFILE  (OpenTopoData ASTER, free, no key) ─────────────
        // ══════════════════════════════════════════════════════════════════════
        let _elevReqId = 0;
        async function loadElevationProfile(routeCoords) {
            const reqId = ++_elevReqId; // version stamp — guard against stale out-of-order responses
            const wrap = document.getElementById('elevation-profile-wrap');
            const canvas = document.getElementById('elev-canvas');
            const loadMsg = document.getElementById('elev-loading-msg');
            if (!wrap || !canvas || !loadMsg || !routeCoords || routeCoords.length < 2) return;

            wrap.style.display = 'flex';
            loadMsg.style.display = 'block';
            canvas.style.display = 'none';
            ['elev-gain','elev-loss','elev-max','elev-min'].forEach(id => { const e = document.getElementById(id); if (e) e.textContent = '…'; });

            try {
                const MAX = 50;
                const step = Math.max(1, Math.floor(routeCoords.length / MAX));
                const sample = [];
                for (let i = 0; i < routeCoords.length; i += step) sample.push(routeCoords[i]);
                if (sample[sample.length-1] !== routeCoords[routeCoords.length-1]) sample.push(routeCoords[routeCoords.length-1]);

                const locStr = sample.map(c => `${c[1].toFixed(5)},${c[0].toFixed(5)}`).join('|');
                const res = await fetch(`https://api.opentopodata.org/v1/aster30m?locations=${locStr}`);
                if (!res.ok) throw new Error(`OpenTopoData ${res.status}`);
                const data = await res.json();
                if (data.status !== 'OK') throw new Error('OpenTopoData error: ' + data.error);
                if (reqId !== _elevReqId) return; // Stale — a newer selection supersedes this response

                const elevFt = data.results.map(r => (r.elevation || 0) * 3.28084);
                const cumDist = [0];
                for (let i = 1; i < sample.length; i++) cumDist.push(cumDist[i-1] + calculateDistanceMeters(sample[i-1], sample[i]));

                loadMsg.style.display = 'none';
                canvas.style.display = 'block';

                const dpr = window.devicePixelRatio || 1;
                const dw = canvas.offsetWidth || 340;
                canvas.width = dw * dpr; canvas.height = 68 * dpr;
                const ctx = canvas.getContext('2d');
                ctx.scale(dpr, dpr);

                const W = dw, H = 68;
                const minE = Math.min(...elevFt), maxE = Math.max(...elevFt);
                const rangeE = Math.max(maxE - minE, 10);
                const totalD = cumDist[cumDist.length - 1];
                const tx = d => (d / totalD) * W;
                const ty = e => H - ((e - minE) / rangeE) * (H * 0.80) - H * 0.08;

                const grad = ctx.createLinearGradient(0, 0, 0, H);
                grad.addColorStop(0, 'rgba(255,51,51,0.55)');
                grad.addColorStop(0.6, 'rgba(255,51,51,0.15)');
                grad.addColorStop(1, 'rgba(255,51,51,0.01)');

                ctx.beginPath();
                ctx.moveTo(tx(cumDist[0]), H);
                elevFt.forEach((e, i) => ctx.lineTo(tx(cumDist[i]), ty(e)));
                ctx.lineTo(tx(totalD), H); ctx.closePath();
                ctx.fillStyle = grad; ctx.fill();

                ctx.beginPath();
                elevFt.forEach((e, i) => i === 0 ? ctx.moveTo(tx(cumDist[i]), ty(e)) : ctx.lineTo(tx(cumDist[i]), ty(e)));
                ctx.strokeStyle = '#ff3333'; ctx.lineWidth = 1.8; ctx.lineJoin = 'round'; ctx.stroke();

                ctx.setLineDash([2,4]); ctx.strokeStyle = 'rgba(255,255,255,0.06)'; ctx.lineWidth = 0.5;
                [0.25,0.5,0.75].forEach(t => { ctx.beginPath(); ctx.moveTo(0,t*H); ctx.lineTo(W,t*H); ctx.stroke(); });
                ctx.setLineDash([]);

                let gain = 0, loss = 0;
                for (let i = 1; i < elevFt.length; i++) { const diff = elevFt[i] - elevFt[i-1]; if (diff > 0) gain += diff; else loss += Math.abs(diff); }
                const fmt = v => v >= 1000 ? `${(v/1000).toFixed(1)}k ft` : `${Math.round(v)} ft`;
                document.getElementById('elev-gain').textContent = '+' + fmt(gain);
                document.getElementById('elev-loss').textContent = '-' + fmt(loss);
                document.getElementById('elev-max').textContent  = fmt(Math.max(...elevFt));
                document.getElementById('elev-min').textContent  = fmt(Math.min(...elevFt));
            } catch(e) {
                loadMsg.textContent = 'Elevation data unavailable';
                canvas.style.display = 'none';
                console.warn('Elevation fetch failed:', e.message);
            }
        }

        let _slAt = 0, _slLoc = null;

        async function updateSpeedLimit(lat, lon) {
            const now = Date.now();
            if (now - _slAt < 15000) return;
            if (_slLoc && calculateDistanceMeters(_slLoc, [lon, lat]) < 80) return;
            _slAt = now; _slLoc = [lon, lat];
            try {
                const q = `[out:json][timeout:8];way["maxspeed"](around:40,${lat.toFixed(5)},${lon.toFixed(5)});out 3;`;
                const res = await fetch('https://overpass-api.de/api/interpreter', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/x-www-form-urlencoded; charset=UTF-8' },
                    body: `data=${encodeURIComponent(q)}`
                });
                if (!res.ok) return;
                const data = await res.json();
                const ways = data.elements || [];
                if (!ways.length) return;
                const raw = ways[0].tags?.maxspeed || '';
                const match = raw.match(/(\d+)/);
                if (!match) return;
                let spd = parseInt(match[1], 10);
                if (!raw.toLowerCase().includes('mph')) spd = Math.round(spd * 0.621371);
                const badge = document.getElementById('speed-limit-beacon');
                const valEl = document.getElementById('sl-value');
                if (badge && valEl) { valEl.textContent = spd; badge.style.display = 'flex'; }
            } catch(e) { /* non-critical */ }
        }
