        // ── Math & Geometry Generators ──────────────────────────────────────
        function createTreePolygon(lon, lat, radiusInDegrees = 0.00003) {
            const points = 6; 
            const coords = [];
            for (let i = 0; i < points; i++) {
                const angle = (i / points) * (2 * Math.PI);
                coords.push([lon + Math.cos(angle) * radiusInDegrees, lat + Math.sin(angle) * radiusInDegrees]);
            }
            coords.push(coords[0]);
            return [coords];
        }

        function bufferLineToPolygon(lineCoords, widthInDegrees = 0.00004) {
            if (lineCoords.length < 2) return null;
            const leftCoords = [];
            const rightCoords = [];
            
            for (let i = 0; i < lineCoords.length - 1; i++) {
                const p1 = lineCoords[i];
                const p2 = lineCoords[i + 1];
                const angle = Math.atan2(p2[1] - p1[1], p2[0] - p1[0]);
                const perp = angle + Math.PI / 2;
                
                const dx = Math.cos(perp) * widthInDegrees;
                const dy = Math.sin(perp) * widthInDegrees;
                
                leftCoords.push([p1[0] + dx, p1[1] + dy]);
                rightCoords.push([p1[0] - dx, p1[1] - dy]);
                
                if (i === lineCoords.length - 2) {
                    leftCoords.push([p2[0] + dx, p2[1] + dy]);
                    rightCoords.push([p2[0] - dx, p2[1] - dy]);
                }
            }
            return [leftCoords.concat(rightCoords.reverse()).concat([leftCoords[0]])];
        }

        function pointToSegmentDistance(p, v, w) {
            const l2 = Math.pow(v[0] - w[0], 2) + Math.pow(v[1] - w[1], 2);
            if (l2 === 0) return { dist: Math.hypot(p[0]-v[0], p[1]-v[1]), snap: v };
            let t = ((p[0] - v[0]) * (w[0] - v[0]) + (p[1] - v[1]) * (w[1] - v[1])) / l2;
            t = Math.max(0, Math.min(1, t));
            const snap = [v[0] + t * (w[0] - v[0]), v[1] + t * (w[1] - v[1])];
            return { dist: Math.hypot(p[0]-snap[0], p[1]-snap[1]), snap: snap };
        }

        function getSnappedCoordinates(lng, lat) {
            if (!snapToRoadActive || roadsGeometryData.length === 0) return [lng, lat];
            
            let minDist = Infinity;
            let bestSnap = [lng, lat];
            const p = [lng, lat];

            for (const segment of roadsGeometryData) {
                const result = pointToSegmentDistance(p, segment[0], segment[1]);
                if (result.dist < minDist) {
                    minDist = result.dist;
                    bestSnap = result.snap;
                }
            }
            if (minDist < 0.0005) return bestSnap;
            return [lng, lat];
        }


        // ── Real Life 3D Buildings & Infrastructure Engine ──────────────────
        async function fetchRealLifeVectorData() {
            const bounds = map.getBounds();
            const west = bounds.getWest();
            const south = bounds.getSouth();
            const east = bounds.getEast();
            const north = bounds.getNorth();

            const query = `[out:json][timeout:25];
                (
                  way["building"](${south},${west},${north},${east});
                  relation["building"](${south},${west},${north},${east});
                  way["highway"](${south},${west},${north},${east});
                  node["natural"="tree"](${south},${west},${north},${east});
                );
                out body; >; out skel qt;`;

            try {
                const response = await fetch('https://overpass-api.de/api/interpreter', {
                    method: 'POST',
                    body: query
                });
                const data = await response.json();
                renderVectorPayload(data);
            } catch (err) {
                console.error("Vector streaming engine degradation:", err);
            }
        }

        function renderVectorPayload(data) {
            const nodes = {};
            if (!data.elements) return;

            data.elements.forEach(el => {
                if (el.type === 'node') nodes[el.id] = [el.lon, el.lat];
            });

            const buildingFeatures = [];
            const roadFeatures = [];
            const bridgeFeatures = [];
            const treeFeatures = [];
            roadsGeometryData = [];
            
            data.elements.forEach(el => {
                // Parse Volumetric Trees
                if (el.type === 'node' && el.tags && el.tags.natural === 'tree') {
                    treeFeatures.push({
                        'type': 'Feature',
                        'properties': {
                            'height': parseFloat(el.tags.height || Math.random() * 5 + 3),
                            'min_height': 0
                        },
                        'geometry': {
                            'type': 'Polygon',
                            'coordinates': createTreePolygon(el.lon, el.lat)
                        }
                    });
                }

                if ((el.type === 'way' || el.type === 'relation') && el.nodes) {
                    const coordinates = el.nodes.map(nId => nodes[nId]).filter(Boolean);
                    if (coordinates.length < 2) return;

                    const tags = el.tags || {};
                    
                    // Parse 3D Extruded Colored Buildings
                    if (tags.building) {
                        let height = parseFloat(tags.height || tags["building:levels"] * 3.5 || 12);
                        let minHeight = parseFloat(tags.min_height || tags["building:min_level"] * 3.5 || 0);
                        let color = tags["building:colour"] || null; 
                        
                        buildingFeatures.push({
                            'type': 'Feature',
                            'properties': {
                                'height': height,
                                'min_height': minHeight,
                                'color': color,
                                'type': 'building'
                            },
                            'geometry': {
                                'type': 'Polygon',
                                'coordinates': [coordinates]
                            }
                        });
                    }
                    
                    // Parse Thick 3D Bridges and Snapping Streets
                    if (tags.highway) {
                        for (let i = 0; i < coordinates.length - 1; i++) {
                            roadsGeometryData.push([coordinates[i], coordinates[i+1]]);
                        }

                        if (tags.bridge === "yes") {
                            const bridgePoly = bufferLineToPolygon(coordinates);
                            if (bridgePoly) {
                                bridgeFeatures.push({
                                    'type': 'Feature',
                                    'properties': { 'height': 8, 'min_height': 5 },
                                    'geometry': { 'type': 'Polygon', 'coordinates': bridgePoly }
                                });
                            }
                        } else {
                            roadFeatures.push({
                                'type': 'Feature',
                                'properties': {},
                                'geometry': { 'type': 'LineString', 'coordinates': coordinates }
                            });
                        }
                    }
                }
            });

            if (map.getSource('osm-buildings')) map.getSource('osm-buildings').setData({ 'type': 'FeatureCollection', 'features': buildingFeatures });
            if (map.getSource('osm-roads')) map.getSource('osm-roads').setData({ 'type': 'FeatureCollection', 'features': roadFeatures });
            if (map.getSource('osm-bridges')) map.getSource('osm-bridges').setData({ 'type': 'FeatureCollection', 'features': bridgeFeatures });
            if (map.getSource('osm-trees')) map.getSource('osm-trees').setData({ 'type': 'FeatureCollection', 'features': treeFeatures });
        }

        // ── POI functional Radar Engine & Unit Transformation Matrix ──────────
        async function runPoiSpatialQuery() {
            const center = map.getCenter();
            const listContainer = document.getElementById('poi-results-list');
            listContainer.innerHTML = '<div class="poi-status-msg">Scanning vector grid...</div>';
            
            clearPoiVisualMarkers();

            const query = `[out:json][timeout:25];
                node["amenity"="${currentPoiCategory}"](around:${currentPoiRadiusMeters},${center.lat},${center.lng});
                out body;`;

            try {
                const response = await fetch('https://overpass-api.de/api/interpreter', {
                    method: 'POST',
                    body: query
                });
                const result = await response.json();
                
                if (!result.elements || result.elements.length === 0) {
                    listContainer.innerHTML = '<div class="poi-status-msg">No entries found within current target horizon.</div>';
                    return;
                }

                listContainer.innerHTML = '';
                
                result.elements.forEach(poi => {
                    const poiCoords = [poi.lon, poi.lat];
                    const name = poi.tags.name || `${currentPoiCategory.toUpperCase()} Node`;
                    
                    const distMeters = calculateDistanceMetric(center.lng, center.lat, poi.lon, poi.lat);
                    const formattedDistance = formatDistanceString(distMeters);

                    const card = document.createElement('div');
                    card.className = 'poi-card';
                    card.innerHTML = `
                        <div class="poi-card-body">
                            <div class="poi-card-name">${name}</div>
                            <div class="poi-card-meta">Distance: ${formattedDistance}</div>
                        </div>
                    `;
                    card.addEventListener('click', () => {
                        map.flyTo({ center: poiCoords, zoom: 17 });
                    });
                    listContainer.appendChild(card);

                    const el = document.createElement('div');
                    el.className = 'live-gps-beacon-node';
                    el.style.backgroundColor = '#00ffcc';
                    el.style.boxShadow = '0 0 12px rgba(0,255,204,0.8)';
                    
                    const marker = new maplibregl.Marker({ element: el })
                        .setLngLat(poiCoords)
                        .setPopup(new maplibregl.Popup({ offset: 10 }).setHTML(`<strong>${name}</strong><br>${formattedDistance}`))
                        .addTo(map);
                        
                    poiMarkersStore.push(marker);
                });

            } catch(err) {
                console.error("POI Grid Query Exception:", err);
                listContainer.innerHTML = '<div class="poi-status-msg">Engine error fetching nearby locations.</div>';
            }
        }

        function calculateDistanceMetric(lon1, lat1, lon2, lat2) {
            const R = 6371e3; 
            const phi1 = lat1 * Math.PI/180;
            const phi2 = lat2 * Math.PI/180;
            const deltaPhi = (lat2-lat1) * Math.PI/180;
            const deltaLambda = (lon2-lon1) * Math.PI/180;
            const a = Math.sin(deltaPhi/2) * Math.sin(deltaPhi/2) +
                      Math.cos(phi1) * Math.cos(phi2) *
                      Math.sin(deltaLambda/2) * Math.sin(deltaLambda/2);
            return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
        }

        function formatDistanceString(meters) {
            if (useImperialUnits) {
                const miles = meters / 1609.344;
                return `${miles.toFixed(2)} mi`;
            } else {
                if (meters >= 1000) {
                    return `${(meters / 1000).toFixed(2)} km`;
                }
                return `${Math.round(meters)} m`;
            }
        }

        function clearPoiVisualMarkers() {
            poiMarkersStore.forEach(m => m.remove());
            poiMarkersStore = [];
        }

        // ── Autonomous Address Resolution Handling Matrix ──────────────────
        function initAutonomousSearchMatrix() {
            const input = document.getElementById('osm-engine-input');
            const dropdown = document.getElementById('osm-engine-dropdown');

            input.addEventListener('input', () => {
                clearTimeout(querySearchDebounceTimer);
                const query = input.value.trim();
                if (query.length < 3) {
                    dropdown.style.display = 'none';
                    return;
                }

                querySearchDebounceTimer = setTimeout(async () => {
                    try {
                        const response = await fetch(`https://nominatim.openstreetmap.org/search?format=json&q=${encodeURIComponent(query)}`);
                        const data = await response.json();
                        
                        dropdown.innerHTML = '';
                        if (!data.length) {
                            dropdown.style.display = 'none';
                            return;
                        }

                        data.slice(0, 5).forEach(item => {
                            const el = document.createElement('div');
                            el.className = 'osm-suggest-item';
                            el.textContent = item.display_name;
                            el.addEventListener('click', () => {
                                targetSelectedCoordinates = [parseFloat(item.lon), parseFloat(item.lat)];
                                dropdown.style.display = 'none';
                                input.value = item.display_name;

                                document.getElementById('context-destination-panel').style.display = 'flex';
                                document.getElementById('node-panel-title').textContent = item.name || "Target Node";
                                document.getElementById('node-panel-address').textContent = item.display_name;

                                if (activeMapMarker) activeMapMarker.remove();
                                
                                const markerEl = document.createElement('div');
                                markerEl.className = 'custom-spatial-marker';
                                markerEl.innerHTML = `<svg width="24" height="24" viewBox="0 0 24 24" fill="none"><circle cx="12" cy="12" r="8" fill="#ff3333" fill-opacity="0.3"/><circle cx="12" cy="12" r="4" fill="#ff3333"/></svg>`;
                                
                                activeMapMarker = new maplibregl.Marker({ element: markerEl })
                                    .setLngLat(targetSelectedCoordinates)
                                    .addTo(map);

                                map.flyTo({ center: targetSelectedCoordinates, zoom: 16 });
                            });
                            dropdown.appendChild(el);
                        });
                        dropdown.style.display = 'block';
                    } catch (err) {
                        console.error("Geocoding Core Failure:", err);
                    }
                }, 400);
            });
        }
