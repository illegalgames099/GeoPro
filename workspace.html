function establishLiveHardwarePositioningStream() { const diagnosticStatus = document.getElementById('telemetry-status-node'); if (!navigator.geolocation) { diagnosticStatus.innerText = "UNSUPPORTED SENSOR"; return; } let initialLockAchieved = false; // ── PASS 1: THE CELLULAR / CELL-TOWER FAST PASS ── // Works instantly on phones via tower/Wi-Fi triangulation. // Bypasses cold hardware GPS spin-up delays. navigator.geolocation.getCurrentPosition( (position) => { if (initialLockAchieved || isNavigating) return; const lon = position.coords.longitude; const lat = position.coords.latitude; currentLiveGpsPosition = [lon, lat]; initialLockAchieved = true; diagnosticStatus.innerText = "ONLINE (CELL INSTANT)"; diagnosticStatus.style.color = "#007aff"; updateUserBeaconOnMap(currentLiveGpsPosition, true); }, (error) => { console.log("Cellular fast-pass unavailable or blocked. Awaiting hardware GPS..."); }, { enableHighAccuracy: false, // Disables satellite requirement for raw speed timeout: 1500, // Hyper-aggressive drop threshold maximumAge: 30000 // Grab cached data up to 30 seconds old } ); // ── PASS 2: THE HIGH-ACCURACY GPS ENGINE ── // Runs simultaneously. If on a laptop/tablet without cell service, // this commands the internal hardware to work at maximum throughput. navigator.geolocation.watchPosition( (position) => { const lon = position.coords.longitude; const lat = position.coords.latitude; if (!isNavigating) { currentLiveGpsPosition = [lon, lat]; // Dynamically tag status based on accuracy radius (under 20m = True Satellites) if (position.coords.accuracy < 20) { diagnosticStatus.innerText = "ONLINE (HARDWARE GPS)"; diagnosticStatus.style.color = "#28a745"; // Success green } else { diagnosticStatus.innerText = "ONLINE (LIVE)"; diagnosticStatus.style.color = "#007aff"; } // Update marker position, only fly camera if it's our first time locking coordinates updateUserBeaconOnMap(currentLiveGpsPosition, !initialLockAchieved); initialLockAchieved = true; } }, (error) => { if (!isNavigating && currentLiveGpsPosition === null) { diagnosticStatus.style.color = "#ff9500"; diagnosticStatus.innerText = "LOCAL EDGE LINK"; } }, { enableHighAccuracy: true, // Forces true hardware GPS/GLONASS utilization timeout: 10000, // Gives slower hardware chips time to find satellites maximumAge: 0 // Prevents stale data fallback loops } ); } // Helper function to manage map marker placement and view snapping function updateUserBeaconOnMap(coords, shouldPanCamera) { if (!userLocationBeaconMarker) { const beaconEl = document.createElement('div'); beaconEl.className = 'live-gps-beacon-node'; userLocationBeaconMarker = new maplibregl.Marker({ element: beaconEl }) .setLngLat(coords) .addTo(map); } else { userLocationBeaconMarker.setLngLat(coords); } if (shouldPanCamera) { map.flyTo({ center: coords, zoom: 14, essential: true }); } }
